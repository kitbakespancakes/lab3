void bubble::Sort(int t)
{

    int k = sizeof(int);
    for(int i = 0; i < k; i++)
    {
        int c[256] = {0};
        for(int j = 0; j < t; j++)
        {
            c[digit(a[j],i)]++;
        }
        for(int j = 1; j < 256;j++)
        {
            c[j]+=c[j-1];
        }
        for(int j = t-1; j >-1;j--)
        {
            y[--c[digit(a[j],i)]] = a[j];
        }
        for(int j = 0; j < t; j++)
        {
            a[j] = y[j];
        }
    }

}
